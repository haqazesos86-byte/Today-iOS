# ============================================================
# Metadata for App Store Connect
# This directory is consumed by `fastlane deliver` / `metadata` lane
# ============================================================
#
# Required structure:
#   metadata/
#   ├── en-US/
#   │   ├── name.txt
#   │   ├── subtitle.txt
#   │   ├── description.txt
#   │   ├── keywords.txt
#   │   ├── promotional_text.txt
#   │   ├── release_notes.txt
#   │   ├── privacy_url.txt
#   │   ├── support_url.txt
#   │   ├── marketing_url.txt
#   │   └── copyright.txt
#   ├── zh-Hans/
#   │   └── ... (same files, in Chinese)
#   ├── copyright.txt
#   ├── primary_category.txt
#   ├── secondary_category.txt  (optional)
#   ├── review_information/
#   │   ├── demo_user.txt
#   │   ├── demo_password.txt
#   │   ├── email_address.txt
#   │   ├── first_name.txt
#   │   ├── last_name.txt
#   │   ├── notes.txt
#   │   └── phone_number.txt
#   └── screenshots/  (auto-generated by `fastlane screenshots`)

# When you have everything ready, run:
#   bundle exec fastlane ios metadata
#
# This will upload all metadata files to App Store Connect.
# Then run:
#   bundle exec fastlane ios beta
# to upload a build to TestFlight.
